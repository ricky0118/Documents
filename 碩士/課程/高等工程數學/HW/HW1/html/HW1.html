function sum = limsum(f,output)
    sum = 0;
    n = 1;
    while 1
        error = f(n);
        sum = sum + error;
        if abs(error) < 1e-6
            break
        end
        n = n+1;
    end
    if output == 1
        fprintf("Iteration times: %d \nValue: %f \nError: %f\n\n", n, sum, error)
    end
end

function dydt = vibration_eqn(t, y)
    global w0 omega
    dydt = [y(2); -w0^2*y(1)+cos(omega*t)];
end
