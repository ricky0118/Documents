%%% ==================== Reward function ====================
% The equation of reward function.
% Author: Wu, Po Hsun
% Date: September 07, 2022
%
\begin{equation} \label{eqn:Reward_function}
    R(t) =
    \begin{cases}
        1, & \text{if}\ -x_\text{bound} \leq x(t) \leq x_\text{bound} \\
        0, & \text{otherwise}
    \end{cases}
\end{equation}
