function dxdt = ODE(~, x)
    % state define: [x_dot, x, lambda1, lambda2]'

    if x(3)<=0
        u = 1;
    else
        u = -1;
    end

    dxdt = [-x(2)+u
             x(1)
            -x(3)+x(4)
             0];
end
