%%% ==================== ANN variable form ====================
% The equation of artificial neural network in variable form.
% Author: Wu, Po Hsun
% Date: May 31, 2022
%
\begin{equation}
    {a^{(1)}} = \sigma \left(
        \mathbf{W}^{(0)} {a^{(0)}}+\mathbf{b}^{(0)}
    \right), \quad
    \begin{cases*}
        \mathbf{W} \in \mathbb{R}^{m\times n} \\
        \mathbf{a} \in \mathbb{R}^{n} \\
        \mathbf{b} \in \mathbb{R}^{m}
    \end{cases*}
    \label{eq:ANN_variable}
\end{equation}
