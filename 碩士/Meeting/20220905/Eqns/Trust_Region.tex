% The equation of Trust region method.
% Author: Wu, Po Hsun
% Date: September 04, 2022
%
\begin{equation}
\begin{aligned}
    \min\ &m_k(p) = f_k+g_k^Tp+\frac{1}{2}p^TB_kp\\
    \text{subject to}\ &\|p\| \leq \Delta_k
\end{aligned}
\end{equation}
